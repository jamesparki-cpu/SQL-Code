CREATE DATABASE sgjpark6; 

USE sgjpark6; 

CREATE TABLE Creator ( 

    crID INT NOT NULL, 

    crName VARCHAR(255), 

    age INT, 

    nationality VARCHAR(255), 

    PRIMARY KEY (crID) 

    ); 

  

CREATE TABLE Studio ( 

    stID INT NOT NULL, 

    stName VARCHAR(255), 

    headquarters VARCHAR(255), 

    PRIMARY KEY (stID) 

); 

  

CREATE TABLE Anime ( 

    anID INT NOT NULL, 

    title VARCHAR(255), 

    genre VARCHAR(100), 

    aFormat VARCHAR(50), 

    PRIMARY KEY (anID) 

); 

  

CREATE TABLE ContractsWith ( 

    crID INT NOT NULL, 

    stID INT NOT NULL, 

    since DATE, 

    PRIMARY KEY (crID, stID), 

    FOREIGN KEY (crID) REFERENCES Creator(crID), 

    FOREIGN KEY (stID) REFERENCES Studio(stID) 

); 

  

CREATE TABLE CreatedBy ( 

    crID INT NOT NULL, 

    anID INT NOT NULL, 

    PRIMARY KEY (crID, anID), 

    FOREIGN KEY (crID) REFERENCES Creator(crID), 

    FOREIGN KEY (anID) REFERENCES Anime(anID) 

    ); 

     

    CREATE TABLE Releases ( 

    stID INT NOT NULL, 

    anID INT NOT NULL, 

    copiesAvailable INT DEFAULT 0, 

    copiesSold INT DEFAULT 0, 

    PRIMARY KEY (stID, anID), 

    FOREIGN KEY (stID) REFERENCES Studio(stID), 

    FOREIGN KEY (anID) REFERENCES Anime(anID) 

);

SELECT DISTINCT s.stName 

FROM Studio s 

JOIN Releases r ON s.stID = r.stID 

JOIN Anime a ON r.anID = a.anID 

WHERE a.aFormat = 'TV'; 


2)

SELECT DISTINCT s.stName 

FROM Studio s 

JOIN Releases r1 ON s.stID = r1.stID 

JOIN Anime a1 ON r1.anID = a1.anID 

JOIN Releases r2 ON s.stID = r2.stID 

JOIN Anime a2 ON r2.anID = a2.anID 

WHERE a1.genre = 'mecha'  

  AND a2.genre = 'slice of life'; 


3)

SELECT s.stName  

FROM Studio s  

JOIN Releases r ON s.stID = r.stID  

JOIN Anime a ON r.anID = a.anID  

WHERE a.genre = 'isekai' AND r.copiesAvailable > 0  

GROUP BY s.stID, s.stName  

HAVING COUNT(DISTINCT a.anID) = (SELECT COUNT(*) FROM Anime WHERE genre = 'isekai'); 


4)

SELECT DISTINCT s.stID 

FROM Studio s 

JOIN Releases r ON s.stID = r.stID 

JOIN Anime a ON r.anID = a.anID 

WHERE a.genre = 'slice of life' 

  AND s.stID NOT IN ( 

      SELECT c_link.stID 

      FROM ContractsWith c_link 

      JOIN Creator c ON c_link.crID = c.crID 

      WHERE c.age > 30 

  );  


5)

SELECT s.stID, MAX(c.age) AS OldestCreatorAge 

FROM Studio s 

JOIN ContractsWith cw ON s.stID = cw.stID 

JOIN Creator c ON cw.crID = c.crID 

GROUP BY s.stID;  


6)

SELECT DISTINCT c.crName 

FROM Creator c 

WHERE c.age > ANY ( 

    SELECT c2.age 

    FROM Creator c2 

    JOIN ContractsWith cw ON c2.crID = cw.crID 

    JOIN Studio s ON cw.stID = s.stID 

    WHERE s.stName = 'Mecha Forge' 

);  


7)

SELECT DISTINCT c.crName 

FROM Creator c 

JOIN ContractsWith cw ON c.crID = cw.crID 

WHERE cw.stID IN ( 

    SELECT r.stID 

    FROM Releases r 

    JOIN Anime a ON r.anID = a.anID 

    GROUP BY r.stID 

    HAVING COUNT(DISTINCT a.genre) >= 5 

);
